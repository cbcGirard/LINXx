elseif cmd == 0x0013 then cmdName= "Get Device User Id"
elseif cmd == 0x0014 then cmdName= "Set Device Ethernet IP"

elseif cmd == 0x0065 then cmdName= "LINXx Analog Write"

elseif cmd == 0x0163 then cmdName= " WS2812 Refresh"

end
